\begin{proposition}[Knuth binomial identity]
  \begin{align*}
    \sum_{k=0}^{n} \binom{n}{k} \binom{r-sk}{n} (-1)^k = s^n
  \end{align*}
\end{proposition}

\begin{proposition}[Falling binomial identity]
  \begin{align*}
    \binom{n}{k} = \frac{\fallingFactorial{n}{k}}{k!}
  \end{align*}
\end{proposition}

\begin{proposition}[Binomial coefficients in stirling form]
  \begin{align*}
    \binom{n}{k} = \frac{\fallingFactorial{n}{k}}{k!}
    &= \frac{1}{k!} \sum_{j=0}^{k} (-1)^{j} \stirlingi{k}{j} n^j \\
    &= \frac{1}{k!} \left(
      \stirlingi{k}{0} n^0 - \stirlingi{k}{1} n^1 + \stirlingi{k}{2} n^2 + \cdots + (-1)^{k} \stirlingi{k}{k} n^k
    \right)
  \end{align*}
\end{proposition}

\begin{corollary}
  \begin{align*}
    \binom{r-sk}{n} = \frac{1}{n!} \sum_{j=0}^{n} (-1)^{j} \stirlingi{n}{j} (r-sk)^j
  \end{align*}
\end{corollary}

\begin{proposition}[Binomial theorem]
  \begin{align*}
    (r-sk)^j
    &= \sum_{t=0}^{j} (-1)^{j-t} \binom{j}{t} r^{j} sk^{j-t} \\
    &= (-1)^{j} (sk)^{j} + \sum_{t=1}^{j} (-1)^{j-t} \binom{j}{t} r^{j} (sk)^{j-t}
  \end{align*}
\end{proposition}
